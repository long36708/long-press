useVirtualList ​

useVirtualList 是一个用于虚拟列表渲染的 Hook。

js
import { useVirtualList } from 'ahooks';

function App() {
	const { list, containerProps, wrapperProps } = useVirtualList(data, {
		itemHeight: 60,
	});

	return (
		<div {...containerProps}>
			<div {...wrapperProps}>
				{list.map((ele, index) => (
					<div key={index} style={{ height: 60 }}>
						Row: {ele.data}
					</div>
				))}
			</div>
		</div>
	);
}
